下面是解密需要的代码,可以放到网站底部或其他适合的地方

<script src="https://mingwusuowei.github.io/media/scripts/jquery-1.11.3.min.js" type="text/javascript"></script>
<script type="text/javascript" src="https://mingwusuowei.github.io/media/scripts/rc4.js"></script>
<script type="text/javascript">
function Decrypt(){
	var constr = $("#Ciphertexts").val();
	var pwdstr = document.getElementById("pwd").value;
	var desstr = CryptoJS.RC4.decrypt(constr, pwdstr).toString(CryptoJS.enc.Utf8);
	$("#contentmw").val(desstr);
	document.getElementById("pwd").style.display="none";
	document.getElementById("GOSubmit").style.display="none";
	document.getElementById("Ciphertext").style.display="block";
}
</script>
